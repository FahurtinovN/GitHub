r = [x[0]] #

    if r[j] != x[i]: # ставим усовие
        r.append(x[i]) #append() добавляет в конец списка элемент, переданный ему в качестве аргумента.
        j += 1 # условие
    i += 1 # условие

def lim_max(nums, limit): #создаем функции
    a = - 1
    for i in nums:
        if i < limit and i > a:     #условие которое ограничивает максимум
            a = i
    return a
